[docs]    def test_uuid_format(self):
        with self.assertRaises(jsonschema.ValidationError):
            jsonschema.validate(
                "not a uuid",
                {"type": "string", "format": "uuid"},
                format_checker=jsonschema.FormatChecker()
            )

        jsonschema.validate(
            str(uuid.uuid4()),
            {"type": "string", "format": "uuid"},
            format_checker=jsonschema.FormatChecker()
        )
